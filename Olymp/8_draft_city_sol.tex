\newcommand{\nanobot}[1]{
	\foreach \x in {1,2,...,8}{
		\draw[red] #1 --++({\x*360/8 + 45/2}:.125);
	}
	\filldraw[red] #1 circle(2pt);
}

\newcommand{\pyr}[1]{
	\pgfmathsetmacro{\m}{#1};
	\foreach \x in {0,1,...,\m}{
		\foreach \y in {0,...,\x}
		\draw(\x, 0) --++(60:{\m-\x});
		\draw(60:\x) --++(-60:\x);
		\draw(60:\x) --++({\m - \x},0);
	}
}
